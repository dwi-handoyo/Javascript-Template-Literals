// Conditionals // Ternary conditional (with "?" sign) const lagu = [ { judul: 'Kau Adalah', penyanyi: 'Isyana Saraswati', feat: 'Rayi Putra' }, { judul: 'Tetap Dalam Jiwa', penyanyi: 'Isyana Saraswati' }, { judul: 'Jalan Bersama', penyanyi: 'Jacko', feat: 'Thor' } ]; const el = `
${lagu.map( l => `
  • Judul: ${l.judul}
  • Penyanyi: ${l.penyanyi} ${l.feat ? `(feat. ${l.feat})` : ''}
`).join('')}
`; console.log(el); document.getElementById("demo").innerHTML = el; // result: // Tag Template - Conditional (Exercise) // * Judul: Kau Adalah // * Penyanyi: Isyana Saraswati (feat. Rayi Putra) // * Judul: Tetap Dalam Jiwa // * Penyanyi: Isyana Saraswati // * Judul: Jalan Bersama // * Penyanyi: Jacko (feat. Thor)
